cmake_minimum_required(VERSION 3.0)
set(BIN server)
project( ${BIN})
set(SOURCE_ROOT ../src)
aux_source_directory(${SOURCE_ROOT} SOURCES)
find_package(Boost REQUIRED system log thread date_time filesystem)
message("Boost_LIBRARIES = ${Boost_LIBRARIES}")
add_executable(${BIN} ${SOURCES})
target_include_directories(${BIN} PUBLIC ${Boost_INCLUDE_DIR})
target_link_libraries( ${BIN} ${Boost_LIBRARIES})
if(WIN32)
 set(CMAKE_CXX_FLAGS "/EHsc -DBOOST_ALL_DYN_LINK -DBOOST_ALL_NO_LIB -D_WIN32_WINNT=0x600")
else()
 set(CMAKE_CXX_FLAGS "-Wall -std=c++0x")
 target_link_libraries( ${BIN} -pthread)
endif()
